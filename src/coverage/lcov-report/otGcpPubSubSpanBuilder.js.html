const { SemanticAttributes } = require('@opentelemetry/semantic-conventions');
const { SpanKind } = require('@opentelemetry/api');
const dayjs = require('dayjs');
 
/**
 * Construct 'CONSUMER'(subscriber) span options following OT specification
 *
 * @param {string} subscription - subscription name
 * @param {object} message - PubSub push-subscription message payload, example:
 * {
 *    "message": {
 *        "attributes": {
 *            "key": "value"
 *        },
 *        "data": "SGVsbG8gQ2xvdWQgUHViL1N1YiEgSGVyZSBpcyBteSBtZXNzYWdlIQ==",
 *        "messageId": "2070443601311540",
 *        "message_id": "2070443601311540",
 *        "publishTime": "2021-02-26T19:13:55.749Z",
 *        "publish_time": "2021-02-26T19:13:55.749Z"
 *    },
 *    "subscription": "projects/myproject/subscriptions/mysubscription"
 * }
 * @returns {object} - OT SpanOptions object
 */
const createSubscriberSpanOptions = (subscription, message) => {
  const attributes = {
    [SemanticAttributes.MESSAGING_SYSTEM]: 'pubsub',
    [SemanticAttributes.MESSAGING_OPERATION]: 'process',
    [SemanticAttributes.MESSAGING_DESTINATION]: subscription,
    [SemanticAttributes.MESSAGING_DESTINATION_KIND]: 'topic',
    [SemanticAttributes.MESSAGING_MESSAGE_ID]: message.messageId,
    [SemanticAttributes.MESSAGING_PROTOCOL]: 'pubsub',
    [SemanticAttributes.MESSAGING_MESSAGE_PAYLOAD_SIZE_BYTES]:
      Buffer.byteLength(message.data ?? '', 'base64'),
    // Not semantic
    'messaging.pubsub.published_at': message.publishTime,
    'messaging.pubsub.received_at': dayjs().utc().format(),
  };
 
  return {
    kind: SpanKind.CONSUMER,
    attributes,
  };
};
 
/**
 * Construct 'PRODUCER'(publisher) span options following OT specification
 *
 * @param {string} topic - name of the topic
 * @param {object} message - full/partial pubsub message: #MessageOptions
 * @returns {object} - OT SpanOptions object
 */
const createPublisherSpanOptions = (topic, message) => {
  const messageData = message.json
    ? JSON.stringify(message.json)
    : message.data ?? '';
 
  const attributes = {
    [SemanticAttributes.MESSAGING_TEMP_DESTINATION]: false,
    [SemanticAttributes.MESSAGING_SYSTEM]: 'pubsub',
    [SemanticAttributes.MESSAGING_OPERATION]: 'send',
    [SemanticAttributes.MESSAGING_DESTINATION]: topic,
    [SemanticAttributes.MESSAGING_DESTINATION_KIND]: 'topic',
    [SemanticAttributes.MESSAGING_PROTOCOL]: 'pubsub',
    [SemanticAttributes.MESSAGING_MESSAGE_PAYLOAD_SIZE_BYTES]:
      Buffer.byteLength(messageData),
    ...(message.orderingKey && {
      'messaging.pubsub.ordering_key': message.orderingKey,
    }),
  };
 
  return {
    kind: SpanKind.PRODUCER,
    attributes,
  };
};
 
module.exports = {
  createPublisherSpanOptions,
  createSubscriberSpanOptions,
};
